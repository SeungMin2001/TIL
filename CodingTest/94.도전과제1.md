```cpp
#include<stdio.h>
int n,A[46],B[46];
int dfs(int x){    //top-down
	if(x==1||x==2) return x;
	if(B[x]==0) B[x]=dfs(x-2)+dfs(x-1);
	return B[x];
}
int main(){
	scanf("%d",&n);
	A[1]=1;
	A[2]=2;
	for(int i=3;i<=n;i++){   //bottom-up
		A[i]=A[i-2]+A[i-1];
	}
	printf("%d %d",A[n],dfs(n));
	
	return 0;
}


```

##### 아까 문제랑 같은 문제이다.
##### 점화식은 n=n-2+n-1이다 바텀업,톱다운 두가지 방식을 구현해봤다.
