```cpp
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
	int t;
	vector<pair<int,int>> v;
	scanf("%d",&t);
	for(int i=0;i<t;i++){
		int x,y;
		scanf("%d %d",&x,&y);
		v.push_back(make_pair(y,x));
	}
	sort(v.begin(),v.end());
	for(int i=0;i<t;i++){
		printf("%d %d\n",v[i].second,v[i].first);
	}
	return 0;
}
```
 ### pair을 통해 벡터를 생성하고 sort하면 first원소 기준으로 sort된다.
