```cpp
#include<stdio.h>
#include<math.h>
using namespace std;
int A[15],cnt=0,N;
bool check(int level){
	for(int i=0;i<level;i++){
		if(A[i]==A[level]||abs(A[level]-A[i])==abs(level-i)) return false;
	}
	return true;
}
int queen_dfs(int level){
	if(level==N){
		cnt++;
		return 0;
	}
	else{
		for(int i=0;i<N;i++){
			A[level]=i;
			if(check(level)==true) queen_dfs(level+1);
		}
	}
	return 0;
}
int main(){
	scanf("%d",&N);
	queen_dfs(0);
	printf("%d",cnt);
	return 0;
}
```
