```cpp
#include<stdio.h>
#include<algorithm>
using namespace std;
int N,M,A[9]={0,},check[10001]={0,},B[10]={0,};
int dfs(int level){
	if(level==M){             //자릿수가 다 채워지면 출력하고 리턴
		for(int i=0;i<M;i++){
			printf("%d ",A[i]);
		}
		printf("\n");
		return 0;
	}
	else{
		for(int i=0;i<N;i++){  //check 배열을 사용해서 같은수를 또 뽑지 못하게 막음
			if(check[i]==0){
				A[level]=B[i];
				check[i]=1;
				dfs(level+1);      // 하나씩 레벨을 올리면서 그 레벨인덱스에 숫자를 넣는 알고리즘
				check[i]=0;
			}
		}
	}
	return 0;
}
int main(){
	scanf("%d %d",&N,&M);
	for(int i=0;i<N;i++){
		scanf("%d",&B[i]);
	}
	sort(B,B+N);    //문제조건에 오름차순으로 넣으라 해서 정렬해줌
	dfs(0);
	return 0;
}
```

### 풀면서 느낀점 : dfs를 이용해서 경우의 수를 구하는 방법만 잘 알고 있으면 쉽게 응용해서 풀수 있는 문제였음
